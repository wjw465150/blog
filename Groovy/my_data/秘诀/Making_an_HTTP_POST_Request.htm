public String urlPost(String charset, String urlStr, int connectTimeout, int readTimeout, String postData) {
  String result=""
  ((HttpURLConnection)new URL(urlStr).openConnection()).with{
    setConnectTimeout(connectTimeout)
    setReadTimeout(readTimeout)

    requestMethod = 'POST'
    doOutput = true
    doInput = true
    setRequestProperty('Content-Type', 'application/json') // Set your content type.
    outputStream.withWriter(charset,{ myWriter  ->
      myWriter.write(postData) // Your post data. Could also use withWriter() if you don't want to write a String.
    })
    // Can check 'responseCode' here if you like.
    result = inputStream.getText(charset) // Using 'inputStream.text' because 'content' will throw an exception when empty.
  }

}
