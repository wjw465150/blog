在groovy1.8中内置了对json格式数据的至此； 
使对json的操作变得非常简捷方便了 

Java代码  
  1. def builder = new JsonBuilder()  
  2. //如同构建对象般  
  3. builder.pepole{  
  4.     person {  
  5.         firstName 'leng'  
  6.         lastName 'feng'  
  7.         //传入map  
  8.         address(  
  9.                 city: 'Shanghai',  
  10.                 country: 'China',  
  11.                 zip: 12345,  
  12.                 )  
  13.         married true  
  14.         //传如list  
  15.         conferences 'JavaOne', 'Gr8conf'  
  16.     }  
  17. }  
  18. //以树形结构输出  
  19. println JsonOutput.prettyPrint(builder.toString())  
  20.   
  21.   
  22. String json = """  
  23. {  
  24.     "pepole": {  
  25.         "person": {  
  26.             "firstName": "leng",  
  27.             "lastName": "feng",  
  28.             "address": {  
  29.                 "city": "Shanghai",  
  30.                 "country": "China",  
  31.                 "zip": 12345  
  32.             },  
  33.             "married": true,  
  34.             "conferences": [  
  35.                 "JavaOne",  
  36.                 "Gr8conf"  
  37.             ]  
  38.         }  
  39.     }  
  40. }  
  41. """  
  42. //类似XmlSlurper  
  43. def root = new JsonSlurper().parseText(json)  
  44. assert root instanceof Map  
  45. assert root.person.conferences instanceof List  
  46. assert root.person.firtsName == 'leng'  
  47. assert root.person.conferences[1] == 'Gr8conf'
