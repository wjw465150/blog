一、构建内部jar
budild.gradle文件

  1. plugins {
  2. //gradle的插件，shadow这个插件可以构建的时候把引用的jar包合进去
  3.     id 'java'
  4.  
  5. //加上此插件后会生成一个shadowJar的task
  6.     id "com.github.johnrengelman.shadow" version "5.0.0"
  7. }
  8.  
  9. version '1.0-SNAPSHOT'
  10.  
  11. sourceCompatibility = 1.8
  12. //构建的jar包的名字
  13. archivesBaseName = 'jar-name'
  14.  
  15. //构建的jar包的版本，会追加到archivesBaseName的后面
  16. version=1.0
  17.  
  18. //引用maven库
  19. repositories {
  20.     mavenCentral()
  21. }
  22. //引用的jar包，在maven库有gradle的引用
  23. dependencies {
  24.     testCompile group: 'junit', name: 'junit', version: '4.12'
  25.  
  26.     compile 'com.zaxxer:HikariCP:3.3.1'
  27.     compile 'mysql:mysql-connector-java:5.1.45'
  28.     compile 'org.apache.logging.log4j:log4j-api:2.11.1'
  29.     compile 'org.apache.logging.log4j:log4j-core:2.11.1'
  30.     compile "org.apache.logging.log4j:log4j-slf4j-impl:2.11.1"
  31. }
  32. //名为shadowJar的task
  33. shadowJar {
  34.     manifest {
  35.  
  36.         //指定入口函数，其他的是默认
  37.         attributes(
  38.                 'Main-Class': 'xx.xx.xx.xx.main'
  39.         )
  40.     }
  41. }

之后执行shadowJar，在build文件下会生成jar包
