利 用gradle shadowjar构建包含依赖的JAR包

正 常情况下，gradle构建的jar包是不包含依赖的，只包含项目源码。

可 是有些情况下我们希望JAR包中也包含相关依赖，譬如说向SPARK集群提交作业的时候，运行JAVA命令的时候。

这 篇文件介绍一种方便的构建assemble jar的方法

  1. apply plugin :'java'
  2. apply plugin :'scala'
  3. apply plugin: 'com.github.johnrengelman.shadow'
  4.  
  5. buildscript {
  6.     repositories {
  7.         jcenter()
  8.     }
  9.     dependencies {
  10.         classpath 'com.github.jengelman.gradle.plugins:shadow:1.2.4'
  11.     }
  12. }
  13.  
  14. repositories {
  15.     mavenCentral()
  16. }
  17.  
  18. shadowJar {
  19.    baseName = 'spark-job'
  20.    classifier = null
  21.    version = null
  22.    dependencies {
  23.       include(dependency('org.apache.flume:flume-ng-core:1.7.0'))
  24.   	  include(dependency('org.apache.spark:spark-streaming-kafka-0-10_2.11:2.0.2'))
  25.   	  include(dependency('org.apache.flume:flume-ng-sdk:1.7.0'))
  26.   	  include(dependency('org.apache.kafka:kafka-clients:0.10.1.0'))
  27.    }
  28. }
  29.  
  30. dependencies {
  31. 	compile group: 'org.apache.spark', name: 'spark-mllib_2.11', version: '2.0.2'
  32. 	compile group: 'org.apache.spark', name: 'spark-streaming_2.11', version: '2.0.2'
  33.     compile group: 'org.apache.spark', name: 'spark-sql_2.11', version: '2.0.2'
  34.     compile group: 'org.apache.spark', name: 'spark-hive_2.11', version: '2.0.2'
  35.     compile 'org.apache.spark:spark-sql-kafka-0-10_2.11:2.0.2'
  36.     compile group: 'org.apache.flume', name: 'flume-ng-core', version: '1.7.0'
  37. 	compile group: 'org.apache.flume', name: 'flume-ng-sdk', version: '1.7.0'
  38.     compile 'org.apache.spark:spark-streaming-kafka-0-10_2.11:2.0.2'
  39.     compile group: 'org.apache.hadoop', name: 'hadoop-common', version: '2.7.3'
  40.     compile group: 'org.apache.hadoop', name: 'hadoop-hdfs', version: '2.7.3'
  41.     compile group: 'org.apache.kafka', name: 'kafka-clients', version: '0.10.1.0'
  42. }


这是某个项目的相关build.gradle

核 心是名为shadow的插件，打包命令为clean shadowJar

你 会发现build好的JAR包中，不仅包含项目源码，还包括flume以及kafka中的类。
