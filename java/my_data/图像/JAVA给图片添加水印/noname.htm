022 /**
023 * 把图片印刷到图片上
024 *
025 * @param pressImg --
026 * 水印文件
027 * @param targetImg --
028 * 目标文件
029 * @param x
030 * --x坐标
031 * @param y
032 * --y坐标
033 */
034 public final static void pressImage(String pressImg, String targetImg,
035 int x, int y) {
036 try {
037 //目标文件
038 File _file = new File(targetImg);
039 Image src = ImageIO.read(_file);
040 int wideth = src.getWidth(null);
041 int height = src.getHeight(null);
042 BufferedImage image = new BufferedImage(wideth, height,
043 BufferedImage.TYPE_INT_RGB);
044 Graphics g = image.createGraphics();
045 g.drawImage(src, 0, 0, wideth, height, null);
046
047 //水印文件
048 File _filebiao = new File(pressImg);
049 Image src_biao = ImageIO.read(_filebiao);
050 int wideth_biao = src_biao.getWidth(null);
051 int height_biao = src_biao.getHeight(null);
052 g.drawImage(src_biao, (wideth - wideth_biao) / 2,
053 (height - height_biao) / 2, wideth_biao, height_biao, null);
054 //水印文件结束
055 g.dispose();
056 FileOutputStream out = new FileOutputStream(targetImg);
057 JPEGImageEncoder encoder = JPEGCodec.createJPEGEncoder(out);
058 encoder.encode(image);
059 out.close();
060 } catch (Exception e) {
061 e.printStackTrace();
062 }
063 }

065 /**
066 * 打印文字水印图片
067 *
068 * @param pressText
069 * --文字
070 * @param targetImg --
071 * 目标图片
072 * @param fontName --
073 * 字体名
074 * @param fontStyle --
075 * 字体样式
076 * @param color --
077 * 字体颜色
078 * @param fontSize --
079 * 字体大小
080 * @param x --
081 * 偏移量
082 * @param y
083 */
084
085 public static void pressText(String pressText, String targetImg,
086 String fontName, int fontStyle, int color, int fontSize, int x,
087 int y) {
088 try {
089 File _file = new File(targetImg);
090 Image src = ImageIO.read(_file);
091 int wideth = src.getWidth(null);
092 int height = src.getHeight(null);
093 BufferedImage image = new BufferedImage(wideth, height,
094 BufferedImage.TYPE_INT_RGB);
095 Graphics g = image.createGraphics();
096 g.drawImage(src, 0, 0, wideth, height, null);
097
098 g.setColor(Color.RED);
099 g.setFont(new Font(fontName, fontStyle, fontSize));
100
101 g.drawString(pressText, wideth - fontSize - x, height - fontSize
102 / 2 - y);
103 g.dispose();
104 FileOutputStream out = new FileOutputStream(targetImg);
105 JPEGImageEncoder encoder = JPEGCodec.createJPEGEncoder(out);
106 encoder.encode(image);
107 out.close();
108 } catch (Exception e) {
109 System.out.println(e);
110 }
111 }
