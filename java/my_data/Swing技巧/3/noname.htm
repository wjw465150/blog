5分钟做一个Dashboard

2010年9月3日 王志远 49 次阅读 发表评论   阅读评论

一大堆图表放一起，综合呈现各种数据统计，估计所有的程序都有这种需求。老外喜欢叫它Dashboard，中文还真没太有合适的对应的词，有人干脆翻译为“仪表盘”，挺怪异的。

有了TWaver的一大堆Chart，做Dashboard就不难了。用GridLayout把一大堆Chart放在一个JPanel中，再做一个Border，放上一些操控的按钮，例如最小化、最大化之类，就有模有样了。

先看这个效果图。

实在没什么技术含量，就废话不多说了，直接上代码。先做一个例子Chart。你可以多用一些TWaver提供的其他各种Chart来丰富视觉效果：

14         this.addXScaleText("Jan");

18         this.addElement(data1, "USD", TWaverUtil.getRandomColor());
19         this.addElement(data2, "RMB", TWaverUtil.getRandomColor());
20   
21         this.addValue(data1, getRandomData(), getRandomData(), getRandomData());

29     private void addElement(Element element, String name, Color color) {

001 public class Test extends JComponent {
002     public Test() {
003         this.setBorder(BorderFactory.createLineBorder(Color.lightGray, 1));
004         this.setLayout(new BorderLayout());
005         this.add(createHeaderPane(), BorderLayout.NORTH);
006         this.add(createChart(), BorderLayout.CENTER);
007     }
008   
009     private JComponent createChart() {
010         return new MyBarChart();
011     }
012   
013     private JPanel createHeaderPane() {
014         JPanel headerPane = new JPanel(new BorderLayout());
015         headerPane.setOpaque(false);
016         headerPane.setBorder(BorderFactory.createEmptyBorder(5, 5, 5, 5));
017         headerPane.add(createNorthPane(), BorderLayout.NORTH);
018   
019         JTextArea txtDescription = new JTextArea("Here is the description for this portlet.");
020         txtDescription.setEditable(false);
021         txtDescription.setEnabled(false);
022         txtDescription.setOpaque(false);
023         txtDescription.setFont(new Font("Dialog", Font.ITALIC, 12));
024         txtDescription.setDisabledTextColor(Color.gray);
025         txtDescription.setLineWrap(true);
026         txtDescription.setBorder(createUnderlineBorder());
027         headerPane.add(txtDescription, BorderLayout.CENTER);
028   
029         return headerPane;
030     }
031   
032     private JPanel createNorthPane() {
033         JPanel northPane = new JPanel(new BorderLayout());
034         northPane.setOpaque(false);
035         JLabel lbTitle = new JLabel("Month Sales");
036         lbTitle.setFont(new Font("Dialog", Font.BOLD, 12));
037         lbTitle.setForeground(new Color(0, 0, 255, 100));
038         northPane.add(lbTitle, BorderLayout.CENTER);
039         JPanel buttonPane = createButtonPane();
040         northPane.add(buttonPane, BorderLayout.EAST);
041         return northPane;
042     }
043   
044     private JPanel createButtonPane() {
045         JPanel buttonPane = new JPanel(new FlowLayout(FlowLayout.RIGHT, 5, 0));
046         buttonPane.setOpaque(false);
047         JButton btnDropDown = new JButton();
048         btnDropDown.setOpaque(false);
049         btnDropDown.setMargin(new Insets(0, 0, 0, 0));
050         btnDropDown.setIcon(TWaverUtil.getIcon("/dashboard/dropdown.png"));
051         btnDropDown.setBorder(null);
052         buttonPane.add(btnDropDown);
053         JButton btnClose = new JButton();
054         btnClose.setMargin(new Insets(0, 0, 0, 0));
055         btnClose.setIcon(TWaverUtil.getIcon("/dashboard/close.png"));
056         btnClose.setBorder(null);
057         btnClose.setOpaque(false);
058         btnClose.addActionListener(new ActionListener() {
059             public void actionPerformed(ActionEvent e) {
060                 Test.this.setVisible(false);
061             }
062         });
063         buttonPane.add(btnClose);
064         return buttonPane;
065     }

067     private static Border createUnderlineBorder() {
068         return new Border() {
069             public void paintBorder(Component c,
070                                     Graphics g,
071                                     int x,
072                                     int y,
073                                     int width,
074                                     int height) {
075                 g.setColor(Color.lightGray);
076                 g.drawLine(x,
077                            y + height - 2,
078                            x + width,
079                            y + height - 2);
080             }
081   
082             public Insets getBorderInsets(Component c) {
083                 return new Insets(0, 0, 2, 0);
084             }
085   
086             public boolean isBorderOpaque() {
087                 return true;
088             }
089         };
090     }
