package thinkinjava.test;

import java.util.logging.Level;
import java.util.logging.Logger;

/**
 *
 * @author Hity
 */
public class GetMethodsTypes {

    public <T> void getInfo(T t){
        System.out.println(t.getClass().getName());
    }
    public static void main(String[] args){
        try {
            Class c = GetMethodsTypes.class;
            GetMethodsTypes gt =(GetMethodsTypes) c.newInstance();
            gt.getInfo(”");
            gt.getInfo(’c');
            gt.getInfo(1);
            gt.getInfo(1111111119999999999L);
            gt.getInfo(gt);
        } catch (InstantiationException ex) {
            Logger.getLogger(GetMethodsTypes.class.getName()).log(Level.SEVERE, null, ex);
        } catch (IllegalAccessException ex) {
            Logger.getLogger(GetMethodsTypes.class.getName()).log(Level.SEVERE, null, ex);
        }
    }
}
泛型方法是在方法的返回值前面用<>里面的参数代替实际的参数。另外think in java里面推荐凡是能用泛型方法解决的问题，一律不要用类泛型。
