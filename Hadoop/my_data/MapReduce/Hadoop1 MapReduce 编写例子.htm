  30. public class WordCount {  
  31.   
  32.     public static class Map extends MapReduceBase implements  
  33.             Mapper<LongWritable, Text, Text, IntWritable> {  
  34.         private final static IntWritable one = new IntWritable(1);  
  35.         private final Text word = new Text();  
  36.   
  37.         /** 
  38.          * map函数继承自MapReduceBase, 并且实现了Mapper接口, 此接口是一个泛型类型,它有4种形式的参数, 
  39.          * 分别用来指定map的输入key类型, 输入value值类型, 输出key值类型和输出value值类型. 
  40.          * 在本例中，输入使用的是TextInputFormat, 它的输出key值是LongWritable类型, 输出value值是Text类型. 
  41.          * 根据本例, 需要输出的是<Text, IntWritable>的形式, 所以输出的key值类型是Text, 
  42.          * 输出的value类型是IntWritable. 
  43.          */  
  44.         @Override  
  45.         public void map(LongWritable key, Text value,  
  46.                 OutputCollector<Text, IntWritable> output, Reporter reporter)  
  47.                 throws IOException {  
  48.             String line = value.toString();  
  49.             StringTokenizer tokenizer = new StringTokenizer(line);  
  50.             while (tokenizer.hasMoreTokens()) {  
  51.                 word.set(tokenizer.nextToken());  
  52.                 output.collect(word, one);  
  53.             }  
  54.   
  55.         }  
  56.     }  
  57.   
  58.     public static class Reduce extends MapReduceBase implements  
  59.             Reducer<Text, IntWritable, Text, IntWritable> {  
  60.   
  61.         /** 
  62.          * reduce函数的输入以map的输出作对应, 因此reduce的输入类型是<Text, IntWritable>. 
  63.          */  
  64.         @Override  
  65.         public void reduce(Text key, Iterator<IntWritable> values,  
  66.                 OutputCollector<Text, IntWritable> output, Reporter reporter)  
  67.                 throws IOException {  
  68.             int sum = 0;  
  69.             while (values.hasNext()) {  
  70.                 sum += values.next().get();  
  71.   
  72.             }  
  73.             output.collect(key, new IntWritable(sum));  
  74.   
  75.         }  
  76.     }
