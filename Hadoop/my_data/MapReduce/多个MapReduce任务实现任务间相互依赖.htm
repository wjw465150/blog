工作过程工遇到有些情况，需要多个mapreduce程序相互依赖执行，使用ControlledJob类可以轻松处理多个JOB之间按照顺序执行。

 

输入文件为2个，t1.txt和t2.txt，内容如下：

t1:

a
b
a
c
d
a
b

 

t2:

c
a
e
d
b
a
e
c
c

 

执行如下命令：

 

hadoop jar /home/sospdm/morejobs.jar morejobs.morejobs input output output1

 

分别得到2个输出文件夹output和output1,内容如下：

 

[sospdm@master2-dev ~]$ hadoop fs -cat output/part-r-00000
a       5
b       3
c       4
d       2
e       2

 

[sospdm@master2-dev ~]$ hadoop fs -cat output1/part-r-00000
sum     16

 

 

 

 

经过处理后第一个MAPREDUCE程序得到词频，第二个程序以第一个输出文件夹作为输入，得出总词频。

代码：

Java代码  
  1. package morejobs;  
  2.   
  3. import java.io.IOException;  
  4. import org.apache.hadoop.fs.Path;  
  5. import org.apache.hadoop.io.IntWritable;  
  6. import org.apache.hadoop.io.Text;  
  7. import org.apache.hadoop.mapred.JobConf;  
  8. import org.apache.hadoop.mapreduce.Job;  
  9. import org.apache.hadoop.mapreduce.Mapper;  
  10. import org.apache.hadoop.mapreduce.Reducer;  
  11. import org.apache.hadoop.mapreduce.lib.input.FileInputFormat;  
  12. import org.apache.hadoop.mapreduce.lib.jobcontrol.ControlledJob;  
  13. import org.apache.hadoop.mapreduce.lib.jobcontrol.JobControl;  
  14. import org.apache.hadoop.mapreduce.lib.output.FileOutputFormat;  
  15. import java.util.regex.Matcher;  
  16. import java.util.regex.Pattern;;  
  17.   
  18.  public class morejobs {  
  19.      
  20. /***************第一个MAPREDUCE实现单词计数 
  21.  *  
  22.  * @author zhangliang 
  23.  * 
  24.  */  
  25.    //第一个Job的map函数  
  26.    public static class Map_First extends Mapper<Object, Text  ,Text , IntWritable>{                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                     
  27.         private final static IntWritable one = new IntWritable(1);  
  28.         private Text keys = new Text();  
  29.         public void map(Object key,Text value, Context context ) throws IOException, InterruptedException {  
  30.             String s = value.toString();  
  31.             keys.set(s);   
  32.             context.write(keys, one);  
  33.         }  
  34.     }  
  35.     
  36.    //第一个Job的reduce函数  
  37.     public static class Reduce_First extends Reducer<Text, IntWritable, Text, IntWritable> {  
  38.       private IntWritable result = new IntWritable();  
  39.       public void reduce(Text key,Iterable<IntWritable>values, Context context) throws IOException, InterruptedException {  
  40.          int sum = 0;  
  41.          for(IntWritable value:values) {  
  42.            sum  +=  value.get();  
  43.          }  
  44.           result.set(sum);  
  45.            
  46.           context.write(key, result);  
  47.       }  
  48.     }  
  49.       
  50.       
  51. /***************************第二个MAPREDUCE实现计数后求和 
  52.  *  
  53.  * @author 11100198 
  54.  * 
  55.  */  
  56.       
  57.       
  58.     //第二个job的map函数  
  59.     public static class Map_Second extends Mapper<Object, Text  ,Text , IntWritable>{                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                                     
  60.         private final static IntWritable one = new IntWritable(1);  
  61.         private Text keys = new Text("sum");  
  62.         public void map(Object key,Text value, Context context ) throws IOException, InterruptedException {  
  63.             Pattern p = Pattern.compile("\\d{1,10}$");  
  64.             Matcher m = p.matcher(value.toString());  
  65.             m.find();  
  66.             int s = Integer.valueOf(m.group()).intValue();  
  67.             one.set(s);  
  68.             context.write(keys, one);  
  69.         }  
  70.     }  
  71.       
  72.     //第二个Job的reduce函数  
  73.     public static class Reduce_Second extends Reducer<Text, IntWritable, Text, IntWritable> {  
  74.       private IntWritable result = new IntWritable();  
  75.       public void reduce(Text key,Iterable<IntWritable>values, Context context) throws IOException, InterruptedException {  
  76.          int sum = 0;  
  77.          for(IntWritable value:values) {  
  78.            sum  +=  value.get();  
  79.          }  
  80.           result.set(sum);  
  81.           context.write(key, result);  
  82.       }  
  83.     }  
  84.   
  85.       
  86. /********************** 
  87.  * 以下JOB配置参考：http://www.tuicool.com/articles/N7buuy 
  88.  * @param args 
  89.  * @throws IOException 
  90.  */  
  91.   
  92.     //启动函数  
  93.     public static void main(String[] args) throws IOException {  
  94.       
  95.     JobConf conf = new JobConf(morejobs.class);  
  96.       
  97.     //第一个job的配置  
  98.     Job job1 = new Job(conf,"join1");  
  99.     job1.setJarByClass(morejobs.class);   
  100.   
  101.       job1.setMapperClass(Map_First.class);   
  102.       job1.setReducerClass(Reduce_First.class);   
  103.   
  104.     job1.setMapOutputKeyClass(Text.class);//map阶段的输出的key   
  105.     job1.setMapOutputValueClass(IntWritable.class);//map阶段的输出的value   
  106.     
  107.     job1.setOutputKeyClass(Text.class);//reduce阶段的输出的key   
  108.     job1.setOutputValueClass(IntWritable.class);//reduce阶段的输出的value   
  109.       
  110.     //加入控制容器   
  111.     ControlledJob ctrljob1=new  ControlledJob(conf);   
  112.     ctrljob1.setJob(job1);   
  113.     //job1的输入输出文件路径  
  114.     FileInputFormat.addInputPath(job1, new Path(args[0]));   
  115.       FileOutputFormat.setOutputPath(job1, new Path(args[1]));   
  116.   
  117.       //第二个作业的配置  
  118.         Job job2=new Job(conf,"Join2");   
  119.       job2.setJarByClass(morejobs.class);   
  120.         
  121.       job2.setMapperClass(Map_Second.class);   
  122.      job2.setReducerClass(Reduce_Second.class);   
  123.        
  124.     job2.setMapOutputKeyClass(Text.class);//map阶段的输出的key   
  125.     job2.setMapOutputValueClass(IntWritable.class);//map阶段的输出的value   
  126.   
  127.     job2.setOutputKeyClass(Text.class);//reduce阶段的输出的key   
  128.     job2.setOutputValueClass(IntWritable.class);//reduce阶段的输出的value   
  129.   
  130.     //作业2加入控制容器   
  131.     ControlledJob ctrljob2=new ControlledJob(conf);   
  132.     ctrljob2.setJob(job2);   
  133.     
  134.      //设置多个作业直接的依赖关系   
  135.        //如下所写：   
  136.      //意思为job2的启动，依赖于job1作业的完成   
  137.     
  138.     ctrljob2.addDependingJob(ctrljob1);   
  139.       
  140.     //输入路径是上一个作业的输出路径，因此这里填args[1],要和上面对应好  
  141.     FileInputFormat.addInputPath(job2, new Path(args[1]));  
  142.       
  143.     //输出路径从新传入一个参数，这里需要注意，因为我们最后的输出文件一定要是没有出现过得  
  144.     //因此我们在这里new Path(args[2])因为args[2]在上面没有用过，只要和上面不同就可以了  
  145.     FileOutputFormat.setOutputPath(job2,new Path(args[2]) );  
  146.   
  147.     //主的控制容器，控制上面的总的两个子作业   
  148.     JobControl jobCtrl=new JobControl("myctrl");   
  149.     
  150.     //添加到总的JobControl里，进行控制  
  151.     jobCtrl.addJob(ctrljob1);   
  152.     jobCtrl.addJob(ctrljob2);   
  153.   
  154.   
  155.     //在线程启动，记住一定要有这个  
  156.     Thread  t=new Thread(jobCtrl);   
  157.     t.start();   
  158.   
  159.     while(true){   
  160.   
  161.     if(jobCtrl.allFinished()){//如果作业成功完成，就打印成功作业的信息   
  162.     System.out.println(jobCtrl.getSuccessfulJobList());   
  163.     jobCtrl.stop();   
  164.     break;   
  165.     }  
  166.     }  
  167.     }  
  168.  }
