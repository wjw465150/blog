#!/bin/sh

# get into run script dir (resolve to absolute path)
SCRIPT_DIR=$(cd $(dirname $0) && pwd)    # This dir is where this script live.
echo "SCRIPT_DIR:$SCRIPT_DIR"
cd $SCRIPT_DIR

#1. SVN:/gfs_v3 on /mnt/nfs_gfs
mountName="/mnt/nfs_gfs"
mountOK=`mount | grep "${mountName}" | wc -l`

if [ 0 -eq $mountOK ]; then
  echo "mount -t nfs SVN:/gfs_v3 ${mountName}"
  #mount -t nfs SVN:/gfs_v3 ${mountName}
  mount -a
  exit 0
else
  echo "${mountName} already mount!"
fi

#2. T1:/gfs_v3 on /data/BAK/gfs_scm_repository
mountName="/data/BAK/gfs_scm_repository"
mountOK=`mount | grep "${mountName}" | wc -l`

if [ 0 -eq $mountOK ]; then
  echo "mount -t nfs SVN:/gfs_v3 ${mountName}"
  #mount -t glusterfs T1:/gfs_v3 ${mountName}
  mount -a
  exit 0
else
  echo "${mountName} already mount!"
fi
